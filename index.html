How it works — signal flow
1. Tap force creates mechanical stress on the PZT disc
2. Crystal lattice deformation separates charges → AC voltage spike (0–90V raw)
3. 1MΩ resistor loads the piezo, dropping voltage to Arduino-safe 0–5V
4. Arduino ADC samples at 10-bit resolution (0–1023)
5. Each LED threshold: ADC value must exceed n×20 for LED n to turn on

The Piezoelectric Effect

Piezoelectricity is the ability of certain crystalline materials — most commonly lead zirconate titanate (PZT), quartz, and PVDF polymer — to generate an electric charge in response to applied mechanical stress. The effect is fully reversible: applying a voltage causes the material to deform.

In your circuit, the disc (likely a 27mm buzzer disc made of brass + PZT ceramic) acts as a transducer. When you press or tap it, the PZT layer bends, separating the internal dipoles. One face becomes net positive, the other net negative — producing a measurable voltage across the leads.

Your Circuit Explained

Your circuit connects a standard 27mm piezo disc to pin A1 of an Arduino Uno, with five LEDs on digital pins 2–6. The code implements a voltage-controlled bar graph (VU meter).

Voltage at Arduino pin:
V_pin = (analogRead(A1) / 1023.0) × 5.0

For raw piezo voltage estimation (with 1MΩ load):
V_est ≈ V_pin × (1,000,000 + R_load) / R_load

Energy per tap estimate:
E = ½ × C × V² (C ≈ 10–20nF for a 27mm disc)

Critical safety note: Piezo discs can spike to ±50V or more. The 1MΩ resistor in parallel "loads" the sensor, clamping the voltage seen by the Arduino ADC pin to safe levels. Without it, you risk permanently damaging the ADC input.

Electricity Generation Potential

A single 27mm PZT disc generates extremely low power — in the microwatt range per tap. The output is inherently AC (spikes positive on press, negative on release). To actually store this energy, you need a bridge rectifier followed by a capacitor or small rechargeable battery.

Arduino sketch — upload this to your Uno 
/*
  PiezoLab — Arduino Uno sketch
  Reads piezo on A1, outputs voltage + raw ADC to Serial.
  Controls 5 LEDs on pins 2–6 as a VU bar graph.
  Serial format: RAW:xxx,VOLTS:y.yyy
  Baud rate: 9600
*/

const int PIEZO_PIN = A1;
const float VREF = 5.0;
const int ADC_MAX = 1023;
const int NUM_LEDS = 5;
int ledPins[NUM_LEDS] = {2, 3, 4, 5, 6};

int peakRaw = 0;
unsigned long peakTimer = 0;

void setup() {
  Serial.begin(9600);
  for (int i = 0; i < NUM_LEDS; i++) {
    pinMode(ledPins[i], OUTPUT);
  }
  Serial.println("PIEZOLAB_READY");
}

void loop() {
  int raw = analogRead(PIEZO_PIN);
  float voltage = (raw / (float)ADC_MAX) * VREF;

  // Track peak (hold for 1 second)
  if (raw > peakRaw) {
    peakRaw = raw;
    peakTimer = millis();
  }
  if (millis() - peakTimer > 1000) {
    peakRaw = 0;
  }

  // Serial output — parsed by Node.js backend
  Serial.print("RAW:"); Serial.print(raw);
  Serial.print(",VOLTS:"); Serial.print(voltage, 3);
  Serial.print(",PEAK:"); Serial.println(peakRaw);

  // LED bar graph
  for (int i = 0; i < NUM_LEDS; i++) {
    digitalWrite(ledPins[i], raw > (i + 1) * 20 ? HIGH : LOW);
  }

  delay(50); // 20 readings/sec
}

// PiezoLab — Node.js backend server
// Run: npm install express socket.io serialport
// Then: node server.js
// Opens on http://localhost:3001

const express = require('express');
const http = require('http');
const { Server } = require('socket.io');
const { SerialPort } = require('serialport');
const { ReadlineParser } = require('@serialport/parser-readline');
const path = require('path');
const fs = require('fs');

const app = express();
const server = http.createServer(app);
const io = new Server(server, {
  cors: { origin: "*" }
});

// Serve the dashboard HTML
app.get('/', (req, res) => {
  res.sendFile(path.join(__dirname, 'index.html'));
});

// ── Auto-detect Arduino serial port ──
async function findArduinoPort() {
  const ports = await SerialPort.list();
  const arduino = ports.find(p =>
    p.manufacturer?.toLowerCase().includes('arduino') ||
    p.manufacturer?.toLowerCase().includes('wch') ||
    p.manufacturer?.toLowerCase().includes('ch340') ||
    p.pnpId?.toLowerCase().includes('arduino')
  );
  return arduino ? arduino.path : ports[0]?.path;
}

let serialPort = null;
let logStream = null;
let tapCount = 0;
let lastRaw = 0;

// Open CSV log file
setupLog();
function setupLog() {
  const fname = `piezo_log_${new Date().toISOString().slice(0,10)}.csv`;
  logStream = fs.createWriteStream(fname, { flags: 'a' });
  logStream.write('timestamp,raw,voltage,peak,taps\n');
  console.log(`Logging to ${fname}`);
}

// Connect to Arduino
async function connectArduino() {
  const portPath = await findArduinoPort();
  if (!portPath) {
    console.log('No serial port found. Running in demo mode.');
    startDemoMode();
    return;
  }
  console.log(`Connecting to ${portPath}...`);
  serialPort = new SerialPort({ path: portPath, baudRate: 9600 });
  const parser = serialPort.pipe(new ReadlineParser({ delimiter: '\n' }));

  parser.on('data', (line) => {
    if (!line.includes('RAW:')) return;
    const raw = parseInt(line.match(/RAW:(\d+)/)?.[1] || 0);
    const volts = parseFloat(line.match(/VOLTS:([\d.]+)/)?.[1] || 0);
    const peak = parseInt(line.match(/PEAK:(\d+)/)?.[1] || 0);
    if (raw > 30 && lastRaw <= 30) tapCount++;
    lastRaw = raw;
    const data = { raw, volts, peak, taps: tapCount, ts: Date.now() };
    io.emit('piezo', data);
    logStream.write(`${new Date().toISOString()},${raw},${volts},${peak},${tapCount}\n`);
  });
  serialPort.on('open', () => { console.log('Arduino connected!'); io.emit('status', 'connected'); });
  serialPort.on('error', (e) => { console.log('Serial error:', e.message); startDemoMode(); });
}

// Demo mode — generates fake data when no Arduino
function startDemoMode() {
  io.emit('status', 'demo');
  setInterval(() => {
    const raw = Math.random() < 0.05 ? Math.floor(Math.random() * 900) : Math.floor(Math.random() * 40);
    const volts = (raw / 1023) * 5;
    if (raw > 30 && lastRaw <= 30) tapCount++;
    lastRaw = raw;
    io.emit('piezo', { raw, volts, peak: raw, taps: tapCount, ts: Date.now() });
  }, 50);
}

io.on('connection', (socket) => {
  console.log('Dashboard connected');
  socket.emit('status', serialPort?.isOpen ? 'connected' : 'demo');
});

server.listen(3001, () => {
  console.log('PiezoLab server running at http://localhost:3001');
  connectArduino();
});
